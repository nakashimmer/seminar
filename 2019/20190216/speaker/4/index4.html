#dlg{
	transform:rotate(0) scale(0);

}

JavaScript

 
BTN1.addEventListener("click",()=>{
	DLG.showModal();
	DLG.style.transform = "rotate(360deg) scale(1)";
});
BTN2.addEventListener("click",()=>{
	DLG.close();
	DLG.style.transform = "rotate(0deg) scale(0)";
});
