const INPUT = document.getElementById("input");
const OBNIZ = new Obniz(obid);

OBNIZ.onconnect = async ()=>{
	const SERVO = OBNIZ.wired(
		"ServoMotor",{gnd:4,vcc:5,signal:6}
	);

	INPUT.addEventListener("change",()=>{
		SERVO.angle(INPUT.value);
	});
}
