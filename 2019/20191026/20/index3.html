円グラフ-JSで操作

CSS

background:conic-gradient(
	green 0 %,
	yellow 0 %,
	blue 0 %,
	red 0 100%
);

JavaScript

 
const G = document.getElementById("green");
const Y = document.getElementById("yellow");
const B = document.getElementById("blue");
const BOX1 = document.getElementById("box1");

G.addEventListener("change", cng);
Y.addEventListener("change", cng);
B.addEventListener("change", cng);

function cng() {
	BOX1.style.background = "conic-gradient(
			green "+G.value+"%,"
			+ "yellow 0 "+Y.value+"%,"
			+ "blue 0 "+B.value+"%,"
			+ "red 0 100%)";
}
