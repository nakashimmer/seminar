			var n1=3;	//ローカル変数を設定
			var n2=5;	//ローカル変数を設定
			n=n1+n2;
			window.alert(n);
		}
		
		function fc1_2(){
			n=n1+n2;
			window.alert(n);
		}	//ローカル変数は影響しない
		
		//---------------------
		function fc2_1(){
			n1=7;		//var がなければグローバル変数
			n2=11;	//var がなければグローバル変数
			window.alert("n1を7に、n2を11に設定しました")
		}
		
		function fc2_2(){
			n=n1+n2;	//グローバル変数での足算になる
			window.alert(n);
		}
