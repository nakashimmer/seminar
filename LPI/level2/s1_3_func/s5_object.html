	<li><button onclick="now();">今の時間</button>
	
	<script>
		function now(){
			var now=new Date();
			var m=now.getMonth();
			var d=now.getDate();
			var h=now.getHours();
			var i=now.getMinutes();
			alert(m+"月"+d+"日"+h+"時"+i+"分");
		}
	</script>
	<!-- 月に注意！ -->
	
	<li><button id="kakudo" onclick="kakudo(90);">sin(90度)</button>
	<script>
	function kakudo(rad){
		window.alert("sin"+rad+"度は"+Math.sin(rad/180*Math.PI));
	}
	</script>
