	<script>
		function now(){
			var now=new Date();
			var m=now.getMonth()+1;
			var d=now.getDate();
			var h=now.getHours();
			var i=now.getMinutes();
			alert(m+"月"+d+"日"+h+"時"+i+"分");
		}
	</script>
